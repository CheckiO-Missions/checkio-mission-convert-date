import datetime


def convert_date(date: str) -> str:
    
    try:
        dd, mm, yyyy = date.split("/")
    except ValueError:
        return "Error: Invalid date."
    
    try:
        D = datetime.date(int(yyyy), int(mm), int(dd))
    except ValueError:
        return "Error: Invalid date."
    
    res = D.isoformat()
    if res!= f"{yyyy}-{mm}-{dd}":
        return "Error: Invalid date."
    
    return res
